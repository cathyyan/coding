## 110. Balanced Binary Tree

```cpp
class Solution {
public:
    bool isBalanced(TreeNode* root, int* height = nullptr) {
        if (!root) {
            if (height) *height = 0;
            return true;
        }
        int lh = 0, rh = 0;
        bool ret = isBalanced(root->left, &lh) &&
                   isBalanced(root->right, &rh) &&
                   std::abs(lh - rh) <= 1;
        if (height) *height = std::max(lh, rh) + 1;
        return ret;
    }
};
```
