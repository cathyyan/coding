class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        for (int i = static_cast<int>(triangle.size()) - 2; i >= 0; --i) {
            for (std::size_t j = 0; j < triangle[i].size(); ++j) {
                triangle[i][j] += std::min(triangle[i + 1][j], triangle[i + 1][j + 1]);
            }
        }
        return triangle[0][0];
    }
};
