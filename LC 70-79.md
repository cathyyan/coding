## 70. Climbing Stairs

Super simple.

```cpp
class Solution {
public:
    std::unordered_map<int, int> dict_;
    
    int climbStairsWithDict(int n) {
        if (dict_.count(n) > 0) return dict_[n];
        return dict_[n] = climbStairs(n);
    }
    
    int climbStairs(int n) {
        if (n < 2) return 1;
        return climbStairsWithDict(n - 1) + climbStairsWithDict(n - 2);
    }
};
```
